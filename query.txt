use db_university;

select *
from degrees;

select *
from students as s
join degrees as d on s.degree_id = d.id
where d.name like "Corso di Laurea in Economia";

select *
from degrees as deg
join departments as dep on deg.department_id = dep.id
where deg.level like "magistrale" and dep.name like "Dipartimento di Neuroscienze";

select t.id, t.name, t.surname, c.*
from teachers as t
join course_teacher as ct on t.id = ct.teacher_id
join courses as c on ct.course_id = c.id
where t.id = 44;

select *
from students as s
join degrees as d on s.degree_id = d.id
join departments as dep on d.department_id = dep.id
order by s.surname, s.name desc;

select *
from degrees as deg
join courses as c on deg.id = c.degree_id
join course_teacher as ct on c.id = ct.course_id
join teachers as t on ct.teacher_id = t.id
order by deg.id, c.id;

select t.name
from teachers as t
join course_teacher as ct on t.id = ct.teacher_id
join courses as c on ct.course_id = c.id
join degrees as deg on c.degree_id = deg.id
join departments as dep on deg.department_id = dep.id
where dep.name like "Dipartimento di Matematica"
group by t.id
